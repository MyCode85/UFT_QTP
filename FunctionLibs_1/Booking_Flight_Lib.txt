Function ReturnOrderNumber()
	strOrderMessage = DataTable.Value("OrderNumber","Global")
	arryOrder = Split(strOrderMessage," ")
	intOrderNumber = arryOrder(1)
	ReturnOrderNumber = intOrderNumber
End Function
